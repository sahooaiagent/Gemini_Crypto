//@version=6
indicator("HILEGA-ALMA", overlay=false, max_bars_back=500, scale=scale.right)

// ─────── TRUE RSI FOR RENKO (Perfect on Renko) ───────
TrueRSI(src, length) =>
    var float up = 0.0
    var float down = 0.0
    delta = src - src[1]
    up := delta > 0 ? delta : 0
    down := delta < 0 ? -delta : 0
    maUp   = ta.alma(up, length, 0.85, 5)
    maDown = ta.alma(down, length, 0.85, 5)
    rs = maUp / maDown
    100 - (100 / (1 + rs))

// ─────── INPUTS ───────
lenRSI   = input.int(11, "RSI Length")
lenVWMA  = input.int(21, "VWMA Period")
TEMA_len = input.int(10, "TEMA Length")

// ─────── CALCULATIONS ───────
rsx     = TrueRSI(close, lenRSI)
vwmaRSX = ta.vwma(rsx, lenVWMA)

// TEMA
e1 = ta.ema(rsx, TEMA_len)
e2 = ta.ema(e1, TEMA_len)
e3 = ta.ema(e2, TEMA_len)
tema = 3 * (e1 - e2) + e3

// ─────── PLOTS (BEAUTIFUL) ───────
plot(rsx,     "True RSI",  color=#00FF00, linewidth=3)
plot(vwmaRSX, "VWMA",      color=#ff0000, linewidth=3)
plot(tema,    "TEMA",      color=#1e00ff, linewidth=3)

hline(90, "Overbought", color=#FF00FF, linestyle=hline.style_dashed)
hline(10, "Oversold",   color=#0066FF, linestyle=hline.style_dashed)
hline(50, "Midline",    color=color.gray)

fill(plot(rsx), plot(vwmaRSX), color=rsx > vwmaRSX ? color.new(#09ff00, 72) : color.new(#ff0000, 68))

// ─────── DIVERGENCE LABELS (100% FIXED – NO ERRORS) ───────
leftBars  = 5
rightBars = 5

// Detect pivots (convert to bool properly)
lowPivot  = ta.pivotlow(rsx, leftBars, rightBars)
highPivot = ta.pivothigh(rsx, leftBars, rightBars)

lowPivotFound  = not na(lowPivot)
highPivotFound = not na(highPivot)

// Regular Bullish Divergence
bullDiv = lowPivotFound and low < low[rightBars*2] and rsx > rsx[rightBars*2]
//lotshape(bullDiv, title="Bull Div", location=location.bottom, color=#00FF00, style=shape.labelup, text="Bull", size=size.small)

// Hidden Bullish Divergence
hiddenBullDiv = lowPivotFound and low > low[rightBars*2] and rsx < rsx[rightBars*2]
//plotshape(hiddenBullDiv, title="H Bull", location=location.bottom, color=#00FFFF, style=shape.labelup, text="H Bull", size=size.small)

// Regular Bearish Divergence
bearDiv = highPivotFound and high > high[rightBars*2] and rsx < rsx[rightBars*2]
//plotshape(bearDiv, title="Bear Div", location=location.top, color=#FF0066, style=shape.labeldown, text="Bear", size=size.small)

// Hidden Bearish Divergence
hiddenBearDiv = highPivotFound and high < high[rightBars*2] and rsx > rsx[rightBars*2]
//plotshape(hiddenBearDiv, title="H Bear", location=location.top, color=#FF00FF, style=shape.labeldown, text="H Bear", size=size.small)

// ─────── BB SQUEEZE (Beautiful Purple) ───────
bbw_val = ta.bbw(close, 20, 2.0)
squeeze = bbw_val == ta.lowest(bbw_val, 125)
bgcolor(squeeze ? color.new(#AA00FF, 90) : na, title="Squeeze")
