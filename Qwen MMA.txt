// ── JMA ───────────────────────────────────────────────────────────────────────
jma_func(src, length, phase, power) =>
    phaseRatio = phase < -100 ? 0.5 : phase > 100 ? 2.5 : phase / 100.0 + 1.5
    beta  = 0.45 * (length - 1) / (0.45 * (length - 1) + 2)
    alpha = math.pow(beta, power)
    var float jma = 0.0
    var float e0  = 0.0
    var float e1  = 0.0
    var float e2  = 0.0
    e0  := (1 - alpha) * src + alpha * nz(e0[1])
    e1  := (src - e0) * (1 - beta) + beta * nz(e1[1])
    e2  := (e0 + phaseRatio * e1 - nz(jma[1])) * math.pow(1 - alpha, 2) + math.pow(alpha, 2) * nz(e2[1])
    jma := e2 + nz(jma[1])
    jma

// ── T3 ────────────────────────────────────────────────────────────────────────
t3_func(src, length, vFactor) =>
    c1 = -(vFactor * vFactor * vFactor)
    c2 =  3 * vFactor * vFactor + 3 * vFactor * vFactor * vFactor
    c3 = -6 * vFactor * vFactor - 3 * vFactor - 3 * vFactor * vFactor * vFactor
    c4 =  1 + 3 * vFactor + vFactor * vFactor * vFactor + 3 * vFactor * vFactor
    e1 = ta.ema(src, length)
    e2 = ta.ema(e1,  length)
    e3 = ta.ema(e2,  length)
    e4 = ta.ema(e3,  length)
    e5 = ta.ema(e4,  length)
    e6 = ta.ema(e5,  length)
    c1 * e6 + c2 * e5 + c3 * e4 + c4 * e3

// ── McGinley ──────────────────────────────────────────────────────────────────
mgd_func(src, length, k) =>
    var float mgd = 0.0
    mgd := na(mgd[1]) ? src : mgd[1] + (src - mgd[1]) / math.max(1, k * length * math.pow(src / mgd[1], 4))
    mgd

// ── KAMA ──────────────────────────────────────────────────────────────────────
kama_func(src, erPeriod, fastEnd, slowEnd) =>
    fastSC   = 2.0 / (fastEnd + 1)
    slowSC   = 2.0 / (slowEnd + 1)
    direction = math.abs(src - src[erPeriod])
    volatility = math.sum(math.abs(src - src[1]), erPeriod)
    er = volatility == 0 ? 0 : direction / volatility
    sc = math.pow(er * (fastSC - slowSC) + slowSC, 2)
    var float kama = na
    kama := na(kama[1]) ? src : kama[1] + sc * (src - kama[1])
    kama

// ── ADX ───────────────────────────────────────────────────────────────────────
calcADX_func(len) =>
    up      = ta.change(high)
    down    = -ta.change(low)
    plusDM  = na(up)   ? na : (up   > down and up   > 0 ? up   : 0)
    minusDM = na(down) ? na : (down > up   and down > 0 ? down : 0)
    trur    = ta.rma(ta.tr, len)
    plus    = fixnan(100 * ta.rma(plusDM,  len) / trur)
    minus   = fixnan(100 * ta.rma(minusDM, len) / trur)
    sum     = plus + minus
    100 * ta.rma(math.abs(plus - minus) / (sum == 0 ? 1 : sum), len)

detectAssetType_func() =>
    sym = syminfo.ticker
    pfx = syminfo.prefix
    isCrypto    = str.contains(sym,"BTC") or str.contains(sym,"ETH") or str.contains(sym,"USDT") or str.contains(pfx,"BINANCE") or str.contains(pfx,"COINBASE") or str.contains(pfx,"CRYPTO")
    isForex     = (str.length(sym) == 6 and not str.contains(sym,"BTC")) or str.contains(pfx,"FX") or str.contains(pfx,"OANDA") or str.contains(pfx,"FOREX")
    isCommodity = str.contains(sym,"GOLD") or str.contains(sym,"OIL") or str.contains(sym,"SILVER") or str.contains(pfx,"COMEX")
    isCrypto ? "Crypto" : isForex ? "Forex" : isCommodity ? "Commodity" : "Stock"

ma_bbUpper      = ma_bbBasis + ma_bbMultiplier * ma_bbDev
ma_bbLower      = ma_bbBasis - ma_bbMultiplier * ma_bbDev

ma_rsi          = ta.rsi(close, ma_manualRsi)

ma_rsiLowest    = ta.lowest(ma_rsi, 50)
ma_rsiHighest   = ta.highest(ma_rsi, 50)
ma_rsiRangeVal  = ma_rsiHighest - ma_rsiLowest
float ma_rsiOversold  = na
float ma_rsiOverbought= na
ma_rsiRange     = 21 - 10  // rsiMax - rsiMin (static for threshold calc)
if ma_enableAdaptive and ma_rsiRangeVal > 20 and ma_rsiRangeVal < 80
    ma_rsiLower = ma_rsiLowest  + ma_rsiRange * 0.2
    ma_rsiUpper = ma_rsiHighest - ma_rsiRange * 0.2
    if ma_regimeIsHighVol
        ma_rsiLower := ma_rsiLower * 0.9
        ma_rsiUpper := ma_rsiUpper * 1.1
    ma_rsiOversold  := math.max(25, math.min(35, ma_rsiLower))
    ma_rsiOverbought:= math.max(65, math.min(75, ma_rsiUpper))
else
    ma_rsiOversold  := 30
    ma_rsiOverbought:= 70

// ═══════════════════════════════════════════════════════════════════════════════
// SECTION 10: STRATEGY MODE + SIGNAL CONDITIONS
// ═══════════════════════════════════════════════════════════════════════════════
ma_vwapValue   = ta.vwap(close)
ma_volumeMA    = ta.sma(volume, 30)
ma_volumeSpike = volume > ma_volumeMA * ma_volumeMultiplier

var string ma_strategyMode = "balanced"
if ma_enableAdaptive
    if ma_regimeIsRanging and ma_regimeIsLowVol
        ma_strategyMode := "mean_reversion"
    else if ma_regimeIsTrending and ma_regimeIsHighVol
        ma_strategyMode := "trend_following"
    else
        ma_strategyMode := "balanced"

ma_longCondition  = ta.crossover(ma_fast,  ma_slow)
ma_shortCondition = ta.crossunder(ma_fast, ma_slow)

if ma_useVolumeFilter
    ma_longCondition  := ma_longCondition  and ma_volumeSpike
    ma_shortCondition := ma_shortCondition and ma_volumeSpike

// Angle filter
ma_fastSlope   = (ma_fast - ma_fast[3]) / 3.0
ma_slowSlope   = (ma_slow - ma_slow[3]) / 3.0
ma_slopeDiff   = ma_fastSlope - ma_slowSlope
ma_crossAngle  = math.abs(math.todegrees(math.atan(ma_slopeDiff * 100)))
ma_tfInMinutes = timeframe.in_seconds() / 60
ma_angleThr    = ma_tfInMinutes <= 15 ? ma_angleMin5m : ma_tfInMinutes <= 30 ? ma_angleMin30m : ma_angleMin1h
if ma_useAngleFilter
    ma_longCondition  := ma_longCondition  and ma_crossAngle >= ma_angleThr
    ma_shortCondition := ma_shortCondition and ma_crossAngle >= ma_angleThr

var int ma_barsSinceLastLong  = 999
var int ma_barsSinceLastShort = 999
if ma_longCondition
    ma_barsSinceLastLong  := 0
else
    ma_barsSinceLastLong  += 1
if ma_shortCondition
    ma_barsSinceLastShort := 0
else
    ma_barsSinceLastShort += 1

ma_longValid  = ma_longCondition  and ma_barsSinceLastLong  >= ma_minBarsBetween
ma_shortValid = ma_shortCondition and ma_barsSinceLastShort >= ma_minBarsBetween

if ma_longValid and ma_shortValid
    if ma_regimeIsBullish
        ma_shortValid := false
    else if ma_regimeIsBearish
        ma_longValid  := false
    else
        if ma_momentum > 0
            ma_shortValid := false
        else
            ma_longValid  := false

// ═══════════════════════════════════════════════════════════════════════════════
// SECTION 11: TRADE ENTRY / EXIT LOGIC (from original TEMA system)
// ═══════════════════════════════════════════════════════════════════════════════
var float lastBuySignalHigh  = na
var int   lastBuySignalBar   = na
var float lastSellSignalLow  = na
var int   lastSellSignalBar  = na

var bool  inLongTrade        = false
var bool  inShortTrade       = false
var float entryPrice         = na
var int   entryBar           = na
var string activeTradeType   = "NONE"
var line  entryLine          = na

var bool  pendingBuy         = false
var int   pendingBuyBar      = na
var float pendingBuyHigh     = na
var bool  pendingSell        = false
var int   pendingSellBar     = na
var float pendingSellLow     = na

var bool  buyEntrySignal     = false
var bool  sellEntrySignal    = false

if ma_longCondition
    lastBuySignalHigh := close
    lastBuySignalBar  := bar_index
    pendingBuy        := true
    pendingBuyBar     := bar_index
    pendingBuyHigh    := close
    if not na(pendingBuyBar)
        line.new(x1=pendingBuyBar, y1=pendingBuyHigh, x2=bar_index, y2=pendingBuyHigh, color=color.new(#00f0ff, 70), width=1, style=line.style_dashed)

if ma_shortCondition
    lastSellSignalLow  := close
    lastSellSignalBar  := bar_index
    pendingSell        := true
    pendingSellBar     := bar_index
    pendingSellLow     := close
    if not na(pendingSellBar)
        line.new(x1=pendingSellBar, y1=pendingSellLow, x2=bar_index, y2=pendingSellLow, color=color.new(#ff00ff, 70), width=1, style=line.style_dashed)

if pendingBuy and not inLongTrade and not inShortTrade
    if close > pendingBuyHigh
        inLongTrade    := true
        activeTradeType:= "LONG"
        entryPrice     := pendingBuyHigh
        entryBar       := bar_index
        buyEntrySignal := true
        entryLine      := line.new(x1=entryBar, y1=entryPrice, x2=entryBar, y2=entryPrice, color=color.green, width=3, style=line.style_solid)
        // FIX 1: anchor Buy label just below the entry candle's low using ATR offset
        label.new(x=entryBar, y=low - ma_atr * 0.3, text=" B ", color=color.green, textcolor=color.white, style=label.style_label_up, size=size.tiny)
        pendingBuy     := false

if pendingSell and not inShortTrade and not inLongTrade
    if close < pendingSellLow
        inShortTrade   := true
        activeTradeType:= "SHORT"
        entryPrice     := pendingSellLow
        entryBar       := bar_index
        sellEntrySignal:= true
        entryLine      := line.new(x1=entryBar, y1=entryPrice, x2=entryBar, y2=entryPrice, color=color.red, width=3, style=line.style_solid)
        // FIX 1: anchor Sell label just above the entry candle's high using ATR offset
        label.new(x=entryBar, y=high + ma_atr * 0.3, text=" S ", color=color.red, textcolor=color.white, style=label.style_label_down, size=size.tiny)
        pendingSell    := false

var float exitPrice = na
var int   exitBar   = na

ma_crossUnder = ta.crossunder(ma_fast, ma_slow)
ma_crossOver  = ta.crossover(ma_fast, ma_slow)

if inLongTrade and ma_crossUnder
    exitPrice       := close
    exitBar         := bar_index
    inLongTrade     := false
    activeTradeType := "NONE"
    entryPrice      := na
    entryBar        := na

if inShortTrade and ma_crossOver
    exitPrice       := close
    exitBar         := bar_index
    inShortTrade    := false
    activeTradeType := "NONE"
    entryPrice      := na
    entryBar        := na

if pendingBuy  and bar_index - pendingBuyBar  > 20
    pendingBuy  := false
if pendingSell and bar_index - pendingSellBar > 20
    pendingSell := false

if buyEntrySignal
    buyEntrySignal  := false
if sellEntrySignal
    sellEntrySignal := false

// SECTION 12: STOP LOSS LINES
// Behaviour:
//   • A new signal (Long or Short) draws its SL line and deletes any prior one.
//   • The line extends rightward every bar (no fixed bar-count expiry).
//   • The line is deleted when EITHER of these events occurs:
//       1. An opposing MA crossover signal fires (new trade direction).
//       2. Price closes BEYOND the SL level (stop triggered).
//   • A small label ("Buy SL" or "Sell SL") is drawn above the line at its
//     start for clarity, and deleted together with the line.
// ═══════════════════════════════════════════════════════════════════════════════
var line   stopLossLine      = na
var label  stopLossLabel     = na
var float  signalEntryPrice  = na
var float  signalStopLoss    = na
var string signalDirection   = "NONE"
var int    signalBarIndex    = na

// ── Helper: cleanly remove the active SL line + label ─────────────────────────
sl_clear() =>
    if not na(stopLossLine)
        line.delete(stopLossLine)
    if not na(stopLossLabel)
        label.delete(stopLossLabel)

// ── New LONG signal: draw Buy SL below entry ───────────────────────────────────
if ma_longCondition and sl_enable
    sl_clear()
    signalEntryPrice := close
    signalStopLoss   := signalEntryPrice - (ma_atr * sl_atrMultiplier)
    signalDirection  := "LONG"
    signalBarIndex   := bar_index
    stopLossLine     := line.new(x1=signalBarIndex, y1=signalStopLoss, x2=signalBarIndex + 1, y2=signalStopLoss, color=sl_lineColor, width=sl_lineWidth, style=line.style_solid)
    stopLossLabel    := label.new(x=signalBarIndex, y=signalStopLoss, text="Buy SL", color=color.new(sl_lineColor, 60), textcolor=color.white, style=label.style_label_up, size=size.tiny)

// ── New SHORT signal: draw Sell SL above entry ─────────────────────────────────
if ma_shortCondition and sl_enable
    sl_clear()
    signalEntryPrice := close
    signalStopLoss   := signalEntryPrice + (ma_atr * sl_atrMultiplier)
    signalDirection  := "SHORT"
    signalBarIndex   := bar_index
    stopLossLine     := line.new(x1=signalBarIndex, y1=signalStopLoss, x2=signalBarIndex + 1, y2=signalStopLoss, color=sl_lineColor, width=sl_lineWidth, style=line.style_solid)
    stopLossLabel    := label.new(x=signalBarIndex, y=signalStopLoss, text="Sell SL", color=color.new(sl_lineColor, 60), textcolor=color.white, style=label.style_label_down, size=size.tiny)

// ── Every bar: extend line rightward and check termination conditions ──────────
if not na(stopLossLine) and sl_enable
    line.set_x2(stopLossLine, bar_index)

    // Condition 1 — opposing signal already handled above (new block replaces old)
    // Condition 2 — price closes beyond the SL level (stop triggered)
    sl_breached = (signalDirection == "LONG"  and close < signalStopLoss) or
                  (signalDirection == "SHORT" and close > signalStopLoss)
    if sl_breached
        sl_clear()
        stopLossLine    := na
        stopLossLabel   := na
        signalDirection := "NONE"
