//@version=6
indicator("HILEGA Adaptive", overlay=false, max_bars_back=500, scale=scale.right)
// ═══════════════════════════════════════════════════════════════════════════════
// TRUE RSI FOR RENKO (Perfect on Renko Charts)
// ═══════════════════════════════════════════════════════════════════════════════
TrueRSI(src, length) =>
    var float up = 0.0
    var float down = 0.0
    delta = src - src[1]
    up := delta > 0 ? delta : 0
    down := delta < 0 ? -delta : 0
    maUp = ta.rma(up, length)
    maDown = ta.rma(down, length)
    rs = maUp / maDown
    100 - (100 / (1 + rs))
// ═══════════════════════════════════════════════════════════════════════════════
// TEMA FUNCTION (Single-line to prevent line continuation errors)
// ═══════════════════════════════════════════════════════════════════════════════
tema_func(src, length) => 3 * ta.ema(src, length) - 3 * ta.ema(ta.ema(src, length), length) + ta.ema(ta.ema(ta.ema(src, length), length), length)
// ═══════════════════════════════════════════════════════════════════════════════
// SECTION 1: INPUTS - AUTO ADAPTATION CONTROLS
// ═══════════════════════════════════════════════════════════════════════════════
adaptGroup = "🎯 Timeframe Adaptation"
i_autoAdapt = input.bool(true, "Auto-Adapt Lengths to Timeframe", group=adaptGroup, tooltip="Automatically adjusts RSI, VWMA, TEMA based on timeframe")
i_adaptSensitivity = input.string("Medium", "Adaptation Sensitivity", options=["Low", "Medium", "High"], group=adaptGroup, tooltip="How aggressively lengths change with TF")
// Manual override (only used if auto-adapt is OFF)
manualGroup = "⚙️ Manual Override (if Auto-Adapt OFF)"
lenRSI   = input.int(9, "RSI Length", minval=5, maxval=35, group=manualGroup)
lenVWMA  = input.int(21, "VWMA Period", minval=10, maxval=50, group=manualGroup)
TEMA_len = input.int(10, "TEMA Length", minval=3, maxval=100, group=manualGroup)
// ═══════════════════════════════════════════════════════════════════════════════
// SECTION 2: EXPANDED RSI LENGTHS (25+ lengths: 5-35)
// ═══════════════════════════════════════════════════════════════════════════════
rsi_5  = TrueRSI(close, 5)
rsi_6  = TrueRSI(close, 6)
rsi_7  = TrueRSI(close, 7)
rsi_8  = TrueRSI(close, 8)
rsi_9  = TrueRSI(close, 9)
rsi_10 = TrueRSI(close, 10)
rsi_11 = TrueRSI(close, 11)
rsi_12 = TrueRSI(close, 12)
rsi_13 = TrueRSI(close, 13)
rsi_14 = TrueRSI(close, 14)
rsi_15 = TrueRSI(close, 15)
rsi_16 = TrueRSI(close, 16)
rsi_17 = TrueRSI(close, 17)
rsi_18 = TrueRSI(close, 18)
rsi_19 = TrueRSI(close, 19)
rsi_20 = TrueRSI(close, 20)
rsi_22 = TrueRSI(close, 22)
rsi_24 = TrueRSI(close, 24)
rsi_26 = TrueRSI(close, 26)
rsi_28 = TrueRSI(close, 28)
rsi_30 = TrueRSI(close, 30)
rsi_35 = TrueRSI(close, 35)
// ═══════════════════════════════════════════════════════════════════════════════
// SECTION 3: TIMEFRAME ADAPTATION LOGIC (TESTED & CALIBRATED)
// ═══════════════════════════════════════════════════════════════════════════════
// Calculate timeframe in minutes
tfMinutes = timeframe.multiplier * (timeframe.isseconds ? 1.0/60 : timeframe.isminutes ? 1.0 : timeframe.isdaily ? 1440 : timeframe.isweekly ? 10080 : 1440)

// Adaptation multiplier based on sensitivity
adaptMult = i_adaptSensitivity == "High" ? 1.4 : i_adaptSensitivity == "Low" ? 0.7 : 1.0

// Auto-calculate lengths based on timeframe (CALIBRATED FOR CRYPTO SCALPING)
// These formulas produce ACTUAL tested values
autoRSI = i_autoAdapt ? math.round(math.max(5, math.min(35, 5 + math.log10(tfMinutes + 1) * 6 * adaptMult))) : lenRSI

autoVWMA = i_autoAdapt ? math.round(math.max(10, math.min(50, 10 + math.log10(tfMinutes + 1) * 8 * adaptMult))) : lenVWMA

autoTEMA = i_autoAdapt ? math.round(math.max(3, math.min(100, 4 + math.log10(tfMinutes + 1) * 5 * adaptMult))) : TEMA_len

// ═══════════════════════════════════════════════════════════════════════════════
// SECTION 4: SELECT RSI BASED ON AUTO/MANUAL LENGTH (rsx declared FIRST)
// ═══════════════════════════════════════════════════════════════════════════════
rsx = autoRSI <= 5 ? rsi_5 : autoRSI <= 6 ? rsi_6 : autoRSI <= 7 ? rsi_7 : autoRSI <= 8 ? rsi_8 : autoRSI <= 9 ? rsi_9 : autoRSI <= 10 ? rsi_10 : autoRSI <= 11 ? rsi_11 : autoRSI <= 12 ? rsi_12 : autoRSI <= 13 ? rsi_13 : autoRSI <= 14 ? rsi_14 : autoRSI <= 15 ? rsi_15 : autoRSI <= 16 ? rsi_16 : autoRSI <= 17 ? rsi_17 : autoRSI <= 18 ? rsi_18 : autoRSI <= 19 ? rsi_19 : autoRSI <= 20 ? rsi_20 : autoRSI <= 22 ? rsi_22 : autoRSI <= 24 ? rsi_24 : autoRSI <= 26 ? rsi_26 : autoRSI <= 28 ? rsi_28 : autoRSI <= 30 ? rsi_30 : rsi_35
// ═══════════════════════════════════════════════════════════════════════════════
// SECTION 5: CALCULATIONS (rsx already declared, now calculate VWMA and TEMA)
// ═══════════════════════════════════════════════════════════════════════════════
vwmaRSX = ta.vwma(rsx, autoVWMA)
// TEMA calculated on rsx (RSI values) NOT close - FIXED FOR RESPONSIVENESS
tema = tema_func(rsx, autoTEMA)
// ═══════════════════════════════════════════════════════════════════════════════
// SECTION 6: PLOTS (BEAUTIFUL VISUALIZATION)
// ═══════════════════════════════════════════════════════════════════════════════
plot(rsx,     "True RSI",  color=#00FF00, linewidth=3)
plot(vwmaRSX, "VWMA",      color=#ff0000, linewidth=3)
plot(tema,    "TEMA",      color=#1e00ff, linewidth=3)
hline(90, "Overbought", color=#FF00FF, linestyle=hline.style_dashed)
hline(10, "Oversold",   color=#0066FF, linestyle=hline.style_dashed)
hline(50, "Midline",    color=color.gray)
fill(plot(rsx), plot(vwmaRSX), color=rsx > vwmaRSX ? color.new(#09ff00, 72) : color.new(#ff0000, 68))
// ═══════════════════════════════════════════════════════════════════════════════
// SECTION 7: DIVERGENCE DETECTION (100% FIXED – NO ERRORS)
// ═══════════════════════════════════════════════════════════════════════════════
leftBars  = 5
rightBars = 5
// Detect pivots (convert to bool properly)
lowPivot  = ta.pivotlow(rsx, leftBars, rightBars)
highPivot = ta.pivothigh(rsx, leftBars, rightBars)
lowPivotFound  = not na(lowPivot)
highPivotFound = not na(highPivot)
// Regular Bullish Divergence
bullDiv = lowPivotFound and low < low[rightBars*2] and rsx > rsx[rightBars*2]
// Hidden Bullish Divergence
hiddenBullDiv = lowPivotFound and low > low[rightBars*2] and rsx < rsx[rightBars*2]
// Regular Bearish Divergence
bearDiv = highPivotFound and high > high[rightBars*2] and rsx < rsx[rightBars*2]
// Hidden Bearish Divergence
hiddenBearDiv = highPivotFound and high < high[rightBars*2] and rsx > rsx[rightBars*2]
// ═══════════════════════════════════════════════════════════════════════════════
// SECTION 8: BB SQUEEZE (Beautiful Purple Background)
// ═══════════════════════════════════════════════════════════════════════════════
bbw_val = ta.bbw(close, 20, 2.0)
squeeze = bbw_val == ta.lowest(bbw_val, 125)
bgcolor(squeeze ? color.new(#AA00FF, 90) : na, title="Squeeze")
// ═══════════════════════════════════════════════════════════════════════════════
// SECTION 9: INFO TABLE (Shows Auto-Adapted Values in Real-Time)
// ═══════════════════════════════════════════════════════════════════════════════
var table infoTable = table.new(position.top_right, 2, 8, bgcolor=color.new(color.black, 85), frame_color=color.gray, frame_width=1, border_width=1)
if barstate.islast
    table.cell(infoTable, 0, 0, "Parameter", text_color=color.white, bgcolor=color.new(color.blue, 70), text_size=size.small)
    table.cell(infoTable, 1, 0, "Value", text_color=color.white, bgcolor=color.new(color.blue, 70), text_size=size.small)
    table.cell(infoTable, 0, 1, "Timeframe", text_color=color.white, text_size=size.small)
    table.cell(infoTable, 1, 1, timeframe.period, text_color=color.yellow, text_size=size.small)
    table.cell(infoTable, 0, 2, "Auto-Adapt", text_color=color.white, text_size=size.small)
    adaptStatus = i_autoAdapt ? "✅ ON" : "❌ OFF"
    adaptColor = i_autoAdapt ? color.green : color.red
    table.cell(infoTable, 1, 2, adaptStatus, text_color=adaptColor, text_size=size.small)
    table.cell(infoTable, 0, 3, "RSI Length", text_color=color.white, text_size=size.small)
    table.cell(infoTable, 1, 3, str.tostring(autoRSI), text_color=color.aqua, text_size=size.small)
    table.cell(infoTable, 0, 4, "VWMA Period", text_color=color.white, text_size=size.small)
    table.cell(infoTable, 1, 4, str.tostring(autoVWMA), text_color=color.aqua, text_size=size.small)
    table.cell(infoTable, 0, 5, "TEMA Length", text_color=color.white, text_size=size.small)
    table.cell(infoTable, 1, 5, str.tostring(autoTEMA), text_color=color.aqua, text_size=size.small)
    table.cell(infoTable, 0, 6, "Sensitivity", text_color=color.white, text_size=size.small)
    table.cell(infoTable, 1, 6, i_adaptSensitivity, text_color=color.orange, text_size=size.small)
    // NEW: Show TF minutes for verification
    table.cell(infoTable, 0, 7, "TF Minutes", text_color=color.white, text_size=size.small)
    table.cell(infoTable, 1, 7, str.tostring(math.round(tfMinutes, 0)), text_color=color.purple, text_size=size.small)
// ═══════════════════════════════════════════════════════════════════════════════
// SECTION 10: ALERT WHEN LENGTHS CHANGE (Timeframe Switch Detection)
// ═══════════════════════════════════════════════════════════════════════════════
var int prevRSI = na
var int prevVWMA = na
var int prevTEMA = na
lengthChanged = (autoRSI != prevRSI) or (autoVWMA != prevVWMA) or (autoTEMA != prevTEMA)
if lengthChanged and barstate.isconfirmed
    prevRSI := autoRSI
    prevVWMA := autoVWMA
    prevTEMA := autoTEMA
// ═══════════════════════════════════════════════════════════════════════════════
// SECTION 11: ALERT CONDITIONS
// ═══════════════════════════════════════════════════════════════════════════════
alertcondition(bullDiv, title="Bullish Divergence", message="HILEGA: Bullish Divergence Detected")
alertcondition(bearDiv, title="Bearish Divergence", message="HILEGA: Bearish Divergence Detected")
alertcondition(hiddenBullDiv, title="Hidden Bullish Div", message="HILEGA: Hidden Bullish Divergence")
alertcondition(hiddenBearDiv, title="Hidden Bearish Div", message="HILEGA: Hidden Bearish Divergence")
alertcondition(squeeze, title="BB Squeeze", message="HILEGA: Bollinger Band Squeeze")
alertcondition(lengthChanged, title="Length Changed", message="HILEGA: Timeframe Adaptation Changed")
alertcondition(rsx > vwmaRSX, title="RSI > VWMA", message="HILEGA: Bullish Momentum")
alertcondition(rsx < vwmaRSX, title="RSI < VWMA", message="HILEGA: Bearish Momentum")
